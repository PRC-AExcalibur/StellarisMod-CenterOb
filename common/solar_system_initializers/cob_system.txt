	class = "sc_black_hole"
	
	flags = { empire_home_system }

	usage = custom_empire
	max_instances = 1
	
	planet = {
		class = "pc_black_hole"
		orbit_distance = 0
		orbit_angle = 0
		size = 30
		has_ring = no
	}
	
	planet = {
		home_planet = yes
		
		class = ideal_planet_class
		orbit_distance = 35
		orbit_angle = random
		size = { min = 16 max = 20 }
		has_ring = no
		
		deposit_blockers = none
		modifiers = none

		init_effect = {
			set_planet_flag = cob_origin_planet
		}
	}

	init_effect = {
		set_spawn_system_batch = begin
		no_scope = {
            spawn_system = {
                min_distance >= 0
                max_distance <= 0
                min_orientation_angle = 0
                max_orientation_angle = 360
                initializer = cob_home_system_init_01
                hyperlane = no
                authorize_spawn_on_galactic_core = yes
            }
        }
		random_system = {
			limit = {
				has_star_flag = cob_flag_0
			}
			save_event_target_as = cob_flag_0
			every_neighbor_system = {
				limit = {
					has_hyperlane_to = prev
				}
				remove_hyperlane = {
					from = this
					to = prev
				}
			}
		}
		set_spawn_system_batch = end
		spawn_megastructure = {
			type = "gateway_final"
			planet = system_star
			orbit_distance = 20
		}
	}
}
